CREATE TABLE DOCUMENTS
(
    DocumentID INT PRIMARY KEY IDENTITY,

    ObjectTypeID INT,

    Title NVARCHAR(500),

    CreatedOn DATETIME DEFAULT GETDATE(),

    CreatedBy INT,

    IsDeleted BIT DEFAULT 0
)
CREATE TABLE METADATA_VALUES
(
    MetadataID INT PRIMARY KEY IDENTITY,

    DocumentID INT,

    PropertyID INT,

    ValueText NVARCHAR(MAX)
)

CREATE TABLE FILES
(
    FileID INT PRIMARY KEY IDENTITY,

    FileName NVARCHAR(500),

    OriginalFileName NVARCHAR(500),

    FileExtension NVARCHAR(50),

    FilePath NVARCHAR(MAX),

    FileSize BIGINT,

    ContentType NVARCHAR(200),

    UploadedBy INT,

    UploadedOn DATETIME DEFAULT GETDATE(),

    IsDeleted BIT DEFAULT 0
)
CREATE TABLE DOCUMENT_FILES
(
    DocumentFileID INT PRIMARY KEY IDENTITY,

    DocumentID INT,

    FileID INT
)
